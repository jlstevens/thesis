Mexican hat¶

In [6]:
def gaussian(x, mean, sigma):
    return np.exp(-np.power(x - mean, 2.) / (2 * np.power(sigma, 2.)))
In [7]:

inh_vals = [(x,-0.2*gaussian(x,0,s2)) for x in xs]
combined = [(x,e+i) for ((x,e),(_,i)) in 
            zip(exc_vals, inh_vals)]
